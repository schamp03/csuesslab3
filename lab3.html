population_data <- population_data %>%
  mutate(
    STATE = str_pad(STATE, 2, pad = "0"),
    COUNTY = str_pad(COUNTY, 3, pad = "0"),
    fips = paste0(STATE, COUNTY)
  ) %>%
  filter(trimws(COUNTY) != "000") %>%  
  select(fips, CTYNAME, contains("2021"))
